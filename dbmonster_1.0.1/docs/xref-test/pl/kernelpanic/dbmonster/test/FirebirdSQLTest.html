79          String user   = "dbmonster";
80          String pass   = "dbmonster";
81          String file   = System.getProperty("dbmonster.firebird.db");
82          String url    = "jdbc:firebirdsql:127.0.0.1:" + file;
83          
84          Properties props = new Properties();
85          props.put("dbmonster.jdbc.driver", driver);
86          props.put("dbmonster.jdbc.url", url);
87          props.put("dbmonster.jdbc.username", user);
88          props.put("dbmonster.jdbc.password", pass);
89          props.put("dbmonster.max-tries", "1000");
90  
91          ConnectionProvider cp = 
92              new DBCPConnectionProvider(driver, url, user, pass);
93  
94          Connection conn = cp.getConnection();
95          Statement stmt = conn.createStatement();
96          
97          try {
98              stmt.executeUpdate("DROP TABLE DEFAULT_TEST");
99          } catch (Exception e) {
100         }
101         try {
102             stmt.executeUpdate("DROP TABLE USERS");
103         } catch (Exception e) {
104         }
105         try {
106             stmt.executeUpdate("DROP TABLE GROUPS");
107         } catch (Exception e) {
108         }
109         try {
110             stmt.executeUpdate(
111                 "CREATE TABLE USERS (id int not null,"
112                 + "login varchar(255) not null, id_group int)");
113             stmt.executeUpdate(
114                 "CREATE TABLE GROUPS (id int not null,"
115                 + "name varchar(255) not null, description BLOB SUB_TYPE 1)");
116             stmt.executeUpdate(
117                 "CREATE TABLE DEFAULT_TEST (id int ,"
118                 + "name varchar(255) not null)");
119             stmt.close();
120             conn.close();
121         } catch (Exception e) {
122             System.out.println(e.getMessage());
123             throw e;
124         }
125 
126         DBMonster dbm = new DBMonster();
127         dbm.setConnectionProvider(cp);
128         dbm.setProperties(props);
129         String u = "/pl/kernelpanic/dbmonster/test/resources/dbmonster-schema.xml";
130         URL ur = getClass().getResource(u);
131         Schema schema = SchemaUtil.loadSchema(ur, dbm.getLogger());
132         dbm.addSchema(schema);
133         try {
134             dbm.doTheJob();
135         } catch (Exception e) {
136             e.printStackTrace();
137             throw e;
138         }
139     }
