62  /***
63   * The string key generator uses sort of string magic known from Perl.
64   * in short it starts from i.e. "aaaaa" and increment it by one:
65   * "aaaab", "aaaac"....
66   *
67   * @author Piotr Maj &lt;piotr.maj@kernelpanic.pl&gt;
68   *
69   * @version $Id: StringKeyGenerator.java,v 1.1 2004/05/22 13:14:15 majek Exp $
70   */
71  public class StringKeyGenerator implements KeyGenerator, Initializable {
72  
73      /***
74       * Default value.
75       */
76      public static final String DEFAULT_VALUE = "aaaaaaaa";
77  
78      /***
79       * The key.
80       */
81      private Key key = null;
82  
83      /***
84       * Holds a list of columns.
85       */
86      private List columns = new ArrayList();
87  
88      /***
89       * Initial value. A special value of 0 (zero) means that
90       * initial value should be taken from the database.
91       */
92      private String startValue = "0";
93  
94      /***
95       * Context.
96       */
97      private DBMonsterContext context = null;
98  
99      /***
100      * Next value.
101      */
102     private String nextValue = null;
103 
104     /***
105      * Chars.
106      */
107     public static final char[] CHARS = new char[26];
108     static {
109         for (int i = 'a'; i <= 'z'; i++) {
110             CHARS[i - 'a'] = (char) i;
111         }
112     }
113 
114     /***
115      * Sets the key.
116      *
117      * @param k the key.
118      */
119     public final void setKey(final Key k) {
120         key = k;
121     }
122 
123     /***
124      * Returns columns.
125      *
126      * @return a list of columns
127      */
128     public final List getColumns() {
129         return columns;
130     }
131 
132     /***
133      * Generates a primary key.
134      *
135      * @return a list of generated <code>Column</code>s.
136      *
137      * @throws Exception if generation fails
138      */
139     public final List generate() throws Exception {
140         Column column = (Column) columns.get(0);
141         if (nextValue == null) {
142             throw new Exception("No more values available (generator: "
143                 + getClass().getName() + ", table: "
144                 + key.getTable().getName() + ").");
145         }
146         column.setValue(nextValue);
147         nextValue = incrementString(nextValue);
148         return columns;
149     }
150 
151     /***
152      * @see Initializable#initialize(DBMonsterContext)
153      */
154     public final void initialize(final DBMonsterContext ctx) throws Exception {
155         context = ctx;
156 
157         Log log = (Log) context.getProperty(DBMonster.LOGGER_KEY);
158         if (log.isDebugEnabled()) {
159             log.debug("Initializing generator <" + getClass().getName() + ">.");
160         }
161 
162         if ("0".equals(startValue)) {
163             ConnectionProvider connProv =
164                 (ConnectionProvider) context.getProperty(
165                     DBMonster.CONNECTION_PROVIDER_KEY);
166             Transaction tx = new Transaction(connProv);
167             try {
168                 tx.begin();
169                 Column column = (Column) columns.get(0);
170                 String sql = "SELECT max(" + column.getName() + ") FROM "
171                     + key.getTable().getName();
172 
173                 if (log.isDebugEnabled()) {
174                     log.debug(sql);
175                 }
176 
177                 ResultSet rs = tx.executeQuery(sql);
178                 if (rs.next()) {
179                     nextValue = rs.getString(1);
180                     if (rs.wasNull() || "".equals(nextValue)) {
181                         nextValue = DEFAULT_VALUE;
182                     }
183                     nextValue = incrementString(nextValue);
184                 } else {
185                     nextValue = DEFAULT_VALUE;
186                 }
187                 tx.commit();
188             } catch (Exception e) {
189                 if (log.isFatalEnabled()) {
190                     log.fatal(e.getMessage());
191                 }
192                 tx.abort();
193                 throw e;
194             } finally {
195                 tx.close();
196             }
197         } else {
198             nextValue = DEFAULT_VALUE;
199         }
200 
201         if (log.isDebugEnabled()) {
202             log.debug("Initialization of generator <"
203                 + getClass().getName() + "> finished.");
204         }
205     }

207     /***
208      * Returns a column name.
209      *
210      * @return the name of the column.
211      */
212     public final String getColumnName() {
213         if (columns.size() == 1) {
214             Column column = (Column) columns.get(0);
215             if (column != null) {
216                 return column.getName();
217             }
218         }
219         return null;
220     }
221 
222     /***
223      * Sets the column name and sets up the column.
224      *
225      * @param name the name of the column.
226      */
227     public final void setColumnName(final String name) {
228         Column column = new Column();
229         column.setName(name);
230         columns = new ArrayList();
231         columns.add(column);
232     }
233 
234     /***
235      * Increments a string.
236      *
237      * @param s string to increment
238      * @return incremented string
239      */
240     private final String incrementString(final String s) {
241         StringBuffer in = new StringBuffer(s);
242         int pos = in.length() - 1;
243         while (pos >= 0) {
244             char c = in.charAt(pos);
245             if (c != 'z') {
246                 in.setCharAt(pos, CHARS[c - 'a' + 1]);
247                 break;
248             } else {
249                 in.setCharAt(pos, 'a');
250             }
251             --pos;
252         }
253         if (pos < 0) {
254             return null; // no more strings
255         } else {
256             return in.toString();
257         }
258     }

267     }

273      */
274     public final void setStartValue(final String sv) {
275         startValue = sv;
276     }
