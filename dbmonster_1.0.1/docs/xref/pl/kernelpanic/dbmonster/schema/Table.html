211     /***
212      * Initializes the table.
213      *
214      * @param ctx dbmonster context.
215      *
216      * @throws Exception if initialization fails
217      *
218      */
219     public final void initialize(final DBMonsterContext ctx) throws Exception {
220         context = ctx;
221 
222         Log log = (Log) context.getProperty(DBMonster.LOGGER_KEY);
223 
224         // get column types hash
225         connectionProvider = (ConnectionProvider) context.getProperty(
226                 DBMonster.CONNECTION_PROVIDER_KEY);
227         Connection conn = null;
228         try {
229             if (log.isDebugEnabled()) {
230                 log.debug("Getting column types.");
231             }
232 
233             conn = connectionProvider.getConnection();
234             DatabaseMetaData metaData = conn.getMetaData();
235             String schemaName = (String) ctx.getProperty("dbmonster.jdbc.schema");
236             ResultSet rs = metaData.getColumns(null, schemaName, name, null);
237 
238             String columnName = null;
239             int columnType = 0;
240             while (rs.next()) {
241                 columnName = rs.getString("COLUMN_NAME");
242                 columnType = rs.getInt("DATA_TYPE");
243                 columnTypes.put(columnName, new Integer(columnType));
244                 if (log.isDebugEnabled()) {
245                     log.debug("Column: " + columnName + ", SQL type: " + columnType);
246                 }
247             }
248         } catch (Exception e) {
249             throw new Exception("Couldn't get column types from database. Buggy JDBC driver?");
250         } finally {
251             if (conn != null) {
252                 conn.close();
253             }
254         }
255         String lng = (String) context.getProperty("dbmonster.max-tries"); //$NON-NLS-1$
256         try {
257             maxTries = Long.parseLong(lng);
258         } catch (Exception e) {
259             if (log.isDebugEnabled()) {
260                 log.debug("Unable to parse dbmonster.max-tries." //$NON-NLS-1$
261                     + " Using default value of 1000."); //$NON-NLS-1$
262             }
263         }
264 
265         progressMonitor =
266             (ProgressMonitor) context.getProperty(
267                 DBMonster.PROGRESS_MONITOR_KEY);
268         if (log.isDebugEnabled()) {
269             log.debug("Initializing table <" + name + ">."); //$NON-NLS-1$ //$NON-NLS-2$
270         }
271 
272         StringBuffer sql = new StringBuffer("INSERT INTO "); //$NON-NLS-1$
273         sql.append(name);
274         sql.append(" ("); //$NON-NLS-1$
275 
276         int count = 0;
277         if (key != null && !key.getDatabaseDefault()) {
278             key.initialize(ctx);
279             List cols = key.getGenerator().getColumns();
280             for (int i = 0; i < cols.size(); i++) {
281                 Column column = (Column) cols.get(i);
282                 if (i != 0) {
283                     sql.append(", "); //$NON-NLS-1$
284                 }
285                 sql.append(column.getName());
286                 ++count;
287             }
288         }
289 
290         Iterator it = columnIterator();
291         if (it.hasNext()) {
292             if (key != null && !key.getDatabaseDefault()) {
293                 sql.append(", "); //$NON-NLS-1$
294             }
295         }
296 
297         boolean first = true;
298         while (it.hasNext()) {
299             Column column = (Column) it.next();
300             column.initialize(ctx);
301             if (!first) {
302                 sql.append(", "); //$NON-NLS-1$
303             }
304             sql.append(column.getName());
305             first = false;
306             ++count;
307         }
308 
309         sql.append(") VALUES ("); //$NON-NLS-1$
310         while (--count >= 0) {
311             sql.append("?"); //$NON-NLS-1$
312             if (count != 0) {
313                 sql.append(", "); //$NON-NLS-1$
314             }
315         }
316         sql.append(")"); //$NON-NLS-1$
317 
318         query = sql.toString();
319 
320         if (log.isDebugEnabled()) {
321             log.debug("Query: " + query); //$NON-NLS-1$
322             log.debug("Initialization of table <" + name + "> finished."); //$NON-NLS-1$ //$NON-NLS-2$
323         }
324     }

326     /***
327      * Generates the table.
328      *
329      * @throws Exception if table generation fails
330      */
331     public final void generate() throws Exception {
332         Log log = (Log) context.getProperty(DBMonster.LOGGER_KEY);
333 
334         if (!generated) {
335 
336             if (log.isInfoEnabled()) {
337                 log.info("Generating table <" + name + ">."); //$NON-NLS-1$ //$NON-NLS-2$
338             }
339 
340             int counter = 0;
341             List keyColumns = null;
342             Transaction tx = null;
343             try {
344                 tx = new Transaction(connectionProvider);
345                 tx.begin();
346                 PreparedStatement pstmt = tx.prepareStatement(query);
347                 int currentTry = 0;
348                 while (counter < rows) {
349                     if (Thread.interrupted()) {
350                         throw new InterruptedException();
351                     }
352                     pstmt.clearParameters();
353                     if (key != null && !key.getDatabaseDefault()) {
354                         if (log.isDebugEnabled()) {
355                             log.debug("Generating key: " + key); //$NON-NLS-1$
356                         }
357                         keyColumns = key.generate();
358                     }
359                     Iterator it = columnIterator();
360                     while (it.hasNext()) {
361                         ((Column) it.next()).generate();
362                     }
363 
364                     if (!monitorSet) {
365                         if (progressMonitor != null) {
366                             progressMonitor.setTableName(name);
367                         }
368                         if (progressMonitor != null) {
369                             progressMonitor.setRowsCount(rows);
370                         }
371                         monitorSet = true;
372                     }
373 
374                     int paramCount = 1;
375                     if (key != null && !key.getDatabaseDefault()) {
376                         it = keyColumns.iterator();
377                         while (it.hasNext()) {
378                             Column column = (Column) it.next();
379                             if (log.isDebugEnabled()) {
380                                 log.debug("Key column parameter " + paramCount //$NON-NLS-1$
381                                     + " is " + column.getValue() + "."); //$NON-NLS-1$ //$NON-NLS-2$
382                             }
383                             Integer columnType = ((Integer) columnTypes.get(column.getName()));
384                             if (columnType != null) {
385                                 pstmt.setObject(paramCount++, column.getValue(), columnType.intValue());
386                             } else {
387                                 pstmt.setObject(paramCount++, column.getValue());
388                             }
389                         }
390                     }
391 
392                     it = columnIterator();
393                     while (it.hasNext()) {
394                         Column column = (Column) it.next();
395                         if (log.isDebugEnabled()) {
396                             log.debug("Column parameter " + paramCount //$NON-NLS-1$
397                                 + " is " + column.getValue() + "."); //$NON-NLS-1$ //$NON-NLS-2$
398                         }
399                         Integer columnType = ((Integer) columnTypes.get(column.getName()));
400                         if (columnType != null) {
401                             pstmt.setObject(paramCount++, column.getValue(), columnType.intValue());
402                         } else {
403                             pstmt.setObject(paramCount++, column.getValue());
404                         }
405                     }
406                     try {
407                         tx.execute();
408                         counter++;
409                         if (progressMonitor != null) {
410                             progressMonitor.rowComplete();
411                         }
412                     } catch (Exception e) {
413                         if (log.isDebugEnabled()) {
414                             log.debug("Execution failed for reason: " //$NON-NLS-1$
415                                 + e.getMessage());
416                        }
417                        currentTry++;
418                        if (currentTry > maxTries) {
419                             throw new Exception("Max tries exceeded! Quiting."); //$NON-NLS-1$
420                         }
421                     }
422                     resetColumns();
423                 }
424                 
425                 tx.commit();
426             } catch (Exception e) {
427                 tx.abort();
428                 throw e;
429             } finally {
430                 tx.close();
431             }
432             generated = true;
433             if (progressMonitor != null) {
434                 progressMonitor.tableComplete();
435             }
436             if (log.isInfoEnabled()) {
437                 log.info("Generation of table <" + name + "> finished."); //$NON-NLS-1$ //$NON-NLS-2$
438             }
439         }
440     }

442     /***
443      * Resets key and all columns.
444      */
445     public final void resetColumns() {
446         Iterator it = columnIterator();
447         while (it.hasNext()) {
448             ((Column) it.next()).reset();
449         }
450     }

452     /***
453      * Resets the entire table and mark it as if it was not generated.
454      */
455     public final void reset() {
456         resetColumns();
457         for (int i = 0; i < columns.size(); i++) {
458             Column c = (Column) columns.get(i);
459             c.getGenerator().reset();
460         }
461         generated = false;
462         monitorSet = false;
463     }
